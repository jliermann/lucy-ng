Use sqlite3 directly:
```python
import sqlite3
conn = sqlite3.connect("data/reference/lucy-ng-derep.db")
cur = conn.cursor()

# Bond distance distribution
cur.execute("SELECT bond_distance, SUM(count), COUNT(*) FROM coupling_path_stats GROUP BY bond_distance ORDER BY bond_distance")
for row in cur.fetchall():
    print(f"  Distance {row[0]}: {row[1]:,} total observations, {row[2]:,} unique pairs")

# Top pairs
cur.execute("SELECT carbon_hose, h_carbon_hose, bond_distance, count FROM coupling_path_stats ORDER BY count DESC LIMIT 20")

# Sparsity
cur.execute("SELECT COUNT(DISTINCT carbon_hose || '|' || h_carbon_hose) FROM coupling_path_stats")
```
