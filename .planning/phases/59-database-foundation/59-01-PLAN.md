def migrate_to_v6(self) -> bool:
    current_version = self.get_schema_version()
    if current_version is None or current_version < 6:
        if current_version is None or current_version < 5:
            self.migrate_to_v5()
        migrate_v5_to_v6(self.connection)
        return True
    return False
